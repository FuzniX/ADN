<script>
      window.onload = function() {
          var fileInput = document.querySelector('input[type="file"]');
          var nofileDiv = document.getElementById('nofile');

          fileInput.addEventListener('change', function() {
              if (fileInput.files.length > 0) {
                  nofileDiv.style.display = 'none'; // Cacher la div si un fichier a été importé
              } else {
                  nofileDiv.style.display = 'block'; // Afficher la div si aucun fichier n'a été importé
              }
          });
      };
  </script>
